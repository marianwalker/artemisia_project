zone_grid <- st_bbox(zone_sf) %>% 
  st_as_stars() %>% 
  st_set_crs(st_crs(zone_sf)) %>% 
  st_crop(zone_sf)
## st_as_s2(): dropping Z and/or M coordinate
soil_krige <- krige(bulk_den ~ 1, 
                    soil_clip,
                    zone_grid,
                    model = soil_vgm_fit)
## [using ordinary kriging]
plot(soil_krige)
